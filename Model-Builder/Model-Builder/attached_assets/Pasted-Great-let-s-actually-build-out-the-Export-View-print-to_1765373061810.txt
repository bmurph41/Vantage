I’ll assume:

Frontend: React + TS, Tailwind, TanStack Query.

You already have useOm(omId) and the OM builder module from Phases 1–2.

I’ll focus on:

1) Export View Component

Create a new file:

frontend/src/modules/om-builder/OmExportPage.tsx

import React, { useMemo } from "react";
import { useParams, Link } from "react-router-dom";
import { useOm } from "./omApi";
import { OmPage, OmBlock, OmBrandingSettings, OmPageLayoutConfig } from "./omTypes";
import { BlockRenderer } from "./BlockRenderer";

type RouteParams = {
  projectId: string;
  omId: string;
};

export const OmExportPage: React.FC = () => {
  const { projectId, omId } = useParams<RouteParams>();
  if (!projectId || !omId) {
    return <div className="p-6 text-red-600">Missing project or OM id in route.</div>;
  }

  const { data, isLoading, error } = useOm(omId);

  const sortedPages: OmPage[] = useMemo(() => {
    if (!data?.pages) return [];
    return [...data.pages].sort((a, b) => (a.orderIndex ?? 0) - (b.orderIndex ?? 0));
  }, [data]);

  const branding: OmBrandingSettings | undefined = data?.om?.settings?.branding;

  if (isLoading) {
    return <div className="p-6">Loading OM export view…</div>;
  }

  if (error || !data) {
    return <div className="p-6 text-red-600">Unable to load OM for export.</div>;
  }

  return (
    <div
      className="min-h-screen bg-slate-100 text-slate-900 print:bg-white"
      style={{
        fontFamily: branding?.fontFamily || "system-ui, -apple-system, BlinkMacSystemFont",
      }}
    >
      {/* Toolbar for screen only */}
      <div className="flex items-center justify-between px-6 py-3 border-b border-slate-300 bg-white shadow-sm print:hidden">
        <div className="flex items-center gap-3">
          <Link
            to={`/projects/${projectId}/oms/${omId}/builder`}
            className="text-sm text-slate-600 hover:text-slate-900"
          >
            ← Back to Builder
          </Link>
          <span className="text-sm text-slate-400">|</span>
          <span className="text-sm font-medium text-slate-700">
            Preview &amp; Export: {data.om.name}
          </span>
        </div>

        <div className="flex items-center gap-3">
          <button
            type="button"
            onClick={() => window.print()}
            className="inline-flex items-center rounded-md border border-slate-300 px-3 py-1.5 text-sm font-medium text-slate-700 hover:bg-slate-50"
          >
            Print / Save as PDF
          </button>
        </div>
      </div>

      {/* Pages container */}
      <div className="flex flex-col items-center py-6 space-y-6 print:py-0">
        {sortedPages.map((page, index) => (
          <ExportPage
            key={page.id}
            page={page}
            blocks={page.blocks || []}
            branding={branding}
            pageNumber={index + 1}
            totalPages={sortedPages.length}
            projectId={projectId}
            omId={omId}
          />
        ))}
      </div>
    </div>
  );
};

interface ExportPageProps {
  page: OmPage;
  blocks: OmBlock[];
  branding?: OmBrandingSettings;
  pageNumber: number;
  totalPages: number;
  projectId: string;
  omId: string;
}

// Single printable page
const ExportPage: React.FC<ExportPageProps> = ({
  page,
  blocks,
  branding,
  pageNumber,
  totalPages,
  projectId,
  omId,
}) => {
  const layout: OmPageLayoutConfig = page.layout || {
    layoutType: "single-column",
    showHeader: true,
    showFooter: true,
    showPageNumber: true,
  };

  // Split blocks into columns if needed
  const leftColumnBlocks: OmBlock[] = [];
  const rightColumnBlocks: OmBlock[] = [];
  const fullWidthBlocks: OmBlock[] = [];

  blocks
    .slice()
    .sort((a, b) => (a.orderIndex ?? 0) - (b.orderIndex ?? 0))
    .forEach((block) => {
      const column = block.style?.column || "auto";
      if (layout.layoutType === "two-column") {
        if (column === "right") {
          rightColumnBlocks.push(block);
        } else if (column === "full") {
          fullWidthBlocks.push(block);
        } else {
          leftColumnBlocks.push(block);
        }
      } else {
        // non two-column layouts: everything is full width stacked
        fullWidthBlocks.push(block);
      }
    });

  const headerHeight = branding?.headerHeight ?? 64;
  const footerHeight = branding?.footerHeight ?? 40;

  return (
    <div
      className="
        mm-print-page
        bg-white shadow print:shadow-none border border-slate-200
        w-[8.5in] min-h-[11in] 
        flex flex-col 
        overflow-hidden
        print:w-auto print:border-0 print:m-0 print:rounded-none
      "
    >
      {/* Header */}
      {layout.showHeader !== false && (
        <div
          className="flex items-center justify-between px-6"
          style={{ height: headerHeight }}
        >
          {branding?.logoUrl ? (
            <div className="flex items-center gap-2">
              <img
                src={branding.logoUrl}
                alt="Logo"
                className="h-8 max-w-[160px] object-contain"
              />
              <span className="text-xs text-slate-400">
                {dataSafeTitle(page.title)}
              </span>
            </div>
          ) : (
            <div className="text-xs font-semibold text-slate-500">
              {dataSafeTitle(page.title)}
            </div>
          )}
          <div
            className="h-6 w-24 rounded-full"
            style={{ backgroundColor: branding?.primaryColor || "#0f766e" }}
          />
        </div>
      )}

      {/* Body */}
      <div
        className="flex-1 px-6 pb-4 pt-2"
        style={{
          backgroundColor: branding?.backgroundColor || "#ffffff",
        }}
      >
        {/* Page title (optional) */}
        <h2 className="text-lg font-semibold text-slate-800 mb-3">
          {page.title}
        </h2>

        {/* Layout variants */}
        {layout.layoutType === "cover" ? (
          <CoverLayout blocks={blocks} branding={branding} />
        ) : layout.layoutType === "hero-with-body" ? (
          <HeroWithBodyLayout
            blocks={blocks}
            branding={branding}
            projectId={projectId}
            omId={omId}
          />
        ) : layout.layoutType === "two-column" ? (
          <TwoColumnLayout
            leftBlocks={leftColumnBlocks}
            rightBlocks={rightColumnBlocks}
            fullWidthBlocks={fullWidthBlocks}
            branding={branding}
            projectId={projectId}
            omId={omId}
          />
        ) : (
          <SingleColumnLayout
            blocks={fullWidthBlocks}
            branding={branding}
            projectId={projectId}
            omId={omId}
          />
        )}
      </div>

      {/* Footer */}
      {layout.showFooter !== false && (
        <div
          className="flex items-center justify-between px-6 text-[10px] text-slate-400 border-t border-slate-200"
          style={{ height: footerHeight }}
        >
          <span>MarinaMatch · Confidential Offering Memorandum</span>
          {layout.showPageNumber !== false && (
            <span>
              Page {pageNumber} of {totalPages}
            </span>
          )}
        </div>
      )}
    </div>
  );
};

// Layout helpers

interface LayoutPropsBase {
  branding?: OmBrandingSettings;
  projectId: string;
  omId: string;
}

const SingleColumnLayout: React.FC<
  { blocks: OmBlock[] } & LayoutPropsBase
> = ({ blocks, branding, projectId, omId }) => (
  <div className="flex flex-col gap-3">
    {blocks.map((block) => (
      <BlockRenderer
        key={block.id}
        block={block}
        projectId={projectId}
        branding={branding}
        context="export"
      />
    ))}
  </div>
);

const TwoColumnLayout: React.FC<
  {
    leftBlocks: OmBlock[];
    rightBlocks: OmBlock[];
    fullWidthBlocks: OmBlock[];
  } & LayoutPropsBase
> = ({ leftBlocks, rightBlocks, fullWidthBlocks, branding, projectId, omId }) => (
  <div className="flex flex-col gap-3">
    {/* Full-width blocks at top */}
    {fullWidthBlocks.map((block) => (
      <BlockRenderer
        key={block.id}
        block={block}
        projectId={projectId}
        branding={branding}
        context="export"
      />
    ))}

    {/* Two-column grid */}
    <div className="grid grid-cols-2 gap-3">
      <div className="flex flex-col gap-3">
        {leftBlocks.map((block) => (
          <BlockRenderer
            key={block.id}
            block={block}
            projectId={projectId}
            branding={branding}
            context="export"
          />
        ))}
      </div>
      <div className="flex flex-col gap-3">
        {rightBlocks.map((block) => (
          <BlockRenderer
            key={block.id}
            block={block}
            projectId={projectId}
            branding={branding}
            context="export"
          />
        ))}
      </div>
    </div>
  </div>
);

const CoverLayout: React.FC<
  { blocks: OmBlock[] } & { branding?: OmBrandingSettings }
> = ({ blocks, branding }) => {
  // For now, treat first text block as title, second as subtitle, rest stacked
  const textBlocks = blocks.filter((b) => b.type === "text");
  const otherBlocks = blocks.filter((b) => b.type !== "text");

  const titleBlock = textBlocks[0];
  const subtitleBlock = textBlocks[1];
  const remainingBlocks = [...textBlocks.slice(2), ...otherBlocks];

  return (
    <div className="flex flex-col h-full">
      <div
        className="flex-1 rounded-lg mb-4 flex flex-col justify-center items-start px-8"
        style={{
          backgroundImage: branding?.logoUrl
            ? undefined
            : undefined, // you can layer a background hero image here later
          backgroundColor: branding?.primaryColor || "#0f766e",
          color: "white",
        }}
      >
        <h1 className="text-3xl font-bold mb-2">
          {titleBlock?.content?.markdown || "Offering Memorandum"}
        </h1>
        {subtitleBlock && (
          <p className="text-sm max-w-md opacity-90">
            {subtitleBlock.content?.markdown}
          </p>
        )}
      </div>
      <div className="flex flex-col gap-2">
        {remainingBlocks.map((block) => (
          <div key={block.id} className="text-xs text-slate-700">
            {/* For cover, we can just dump raw markdown/plain text */}
            {block.content?.markdown}
          </div>
        ))}
      </div>
    </div>
  );
};

const HeroWithBodyLayout: React.FC<
  { blocks: OmBlock[] } & LayoutPropsBase
> = ({ blocks, branding, projectId, omId }) => {
  const heroBlocks = blocks.slice(0, 1); // first block as hero text section
  const bodyBlocks = blocks.slice(1);

  return (
    <div className="flex flex-col gap-4">
      <div
        className="rounded-lg px-6 py-4 flex flex-col justify-center"
        style={{
          backgroundColor: branding?.primaryColor || "#0f766e",
          color: "white",
        }}
      >
        {heroBlocks.map((block) => (
          <BlockRenderer
            key={block.id}
            block={block}
            projectId={projectId}
            branding={branding}
            context="export"
          />
        ))}
      </div>
      <div className="flex flex-col gap-3">
        {bodyBlocks.map((block) => (
          <BlockRenderer
            key={block.id}
            block={block}
            projectId={projectId}
            branding={branding}
            context="export"
          />
        ))}
      </div>
    </div>
  );
};

function dataSafeTitle(title?: string | null) {
  if (!title) return "Offering Memorandum";
  return title;
}

3) Wire the Export Route into Your Router

If you’re using React Router v6, add a route in your main router file (e.g., App.tsx):

import { OmExportPage } from "./modules/om-builder/OmExportPage";

// inside <Routes>:
<Route
  path="/projects/:projectId/oms/:omId/export"
  element={<OmExportPage />}
/>


If you’re using Wouter, you’d do something like:

import { Route } from "wouter";
import { OmExportPage } from "./modules/om-builder/OmExportPage";

<Route path="/projects/:projectId/oms/:omId/export" component={OmExportPage} />;

5) Tailwind / CSS for Print

You need a small bit of global CSS so each .mm-print-page becomes a separate PDF page when printing.

If you have a global CSS file (e.g. src/index.css or src/styles.css), add:

@media print {
  body {
    margin: 0;
    padding: 0;
  }

  .mm-print-page {
    /* Each page starts on a new sheet */
    page-break-after: always;
    break-after: page;
  }

  /* Hide the Replit UI or app chrome if necessary */
  .print:hidden {
    display: none !important;
  }
}


Tailwind already lets you use print:* variants (like print:hidden in the toolbar).
