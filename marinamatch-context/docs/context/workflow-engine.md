# MarinaMatch — Workflow Automation Engine

## Architecture Overview

The workflow engine is **stateless** — the rule evaluator has no internal state.
All state lives in the database. The evaluator is called, runs, writes results to DB, and exits.

```
Trigger Event
      ↓
Rule Evaluator (stateless)
  - Load rules for org
  - Check trigger type match
  - Evaluate conditions against event data
  - Execute matched actions
      ↓
Action Executors (7 types)
  - Write results to DB
  - Emit notifications
```

## Trigger Types (7)

```typescript
type TriggerType =
  | 'deal_stage_changed'       // Deal moves to a new pipeline stage
  | 'deal_created'             // New deal created
  | 'deal_value_threshold'     // Deal value crosses a configured threshold
  | 'task_overdue'             // Task passes its due date without completion
  | 'task_completed'           // Task marked complete
```

## Condition Evaluation

Conditions are stored as JSONB arrays. Each condition is evaluated against trigger data.

### Condition Schema
```typescript
interface Condition {
  field: string;           // dot-notation path into trigger data: 'data.newStageName'
  operator: ConditionOperator;
  value: any;
}

type ConditionOperator =
  | 'eq'          // equals
  | 'neq'         // not equals
  | 'gt'          // greater than
  | 'gte'         // greater than or equal
  | 'lt'          // less than
  | 'lte'         // less than or equal
  | 'contains'    // string contains
  | 'in'          // value is in array
  | 'not_in';     // value is not in array
```

### Condition Evaluator
```typescript
function evaluateConditions(
  conditions: Condition[],
  triggerData: TriggerEvent
): boolean {
  // All conditions must pass (AND logic)
  return conditions.every(condition => {
    const value = getNestedValue(triggerData, condition.field);
    return evaluateSingleCondition(value, condition.operator, condition.value);
  });
}

function getNestedValue(obj: any, path: string): any {
  return path.split('.').reduce((current, key) => current?.[key], obj);
}
```

## Rule Evaluator — Core Function

```typescript
// Called whenever a trigger event fires
export async function evaluateRules(
  triggerEvent: TriggerEvent
): Promise<void> {
  const { orgId, type: triggerType } = triggerEvent;

  // Load active rules for this org and trigger type
  const rulesResult = await pool.query(
    `SELECT * FROM workflow_rules
     WHERE org_id = $1
       AND trigger_type = $2
       AND is_active = true
     ORDER BY priority DESC`,
    [orgId, triggerType]
  );

  for (const ruleRow of rulesResult.rows) {
    const rule = mapRuleRow(ruleRow);

    // Evaluate conditions
    const conditionsMet = evaluateConditions(rule.conditions, triggerEvent);

    // Log execution (even if conditions not met)
    const execResult = await pool.query(
      `INSERT INTO workflow_executions
       (org_id, rule_id, trigger_type, trigger_data, conditions_met, status)
       VALUES ($1, $2, $3, $4, $5, $6)
       RETURNING id`,
      [orgId, rule.id, triggerType, JSON.stringify(triggerEvent), conditionsMet, 'pending']
    );
    const executionId = execResult.rows[0].id;

    if (!conditionsMet) continue;

    // Execute actions
    const executedActions = [];
    for (const action of rule.actions) {
      try {
        await executeAction(action, triggerEvent, executionId, orgId);
        executedActions.push({ type: action.type, status: 'success' });
      } catch (err) {
        executedActions.push({ type: action.type, status: 'failed', error: String(err) });
      }
    }

    // Update execution record
    await pool.query(
      `UPDATE workflow_executions
       SET status = 'completed', actions_executed = $1
       WHERE id = $2`,
      [JSON.stringify(executedActions), executionId]
    );
  }
}
```

## Token Substitution in Actions

Actions support `{{TOKEN}}` substitution from trigger event data:

```typescript
function substituteTokens(template: string, triggerEvent: TriggerEvent): string {
  return template.replace(/\{\{([A-Z_]+)\}\}/g, (match, token) => {
    const tokenMap: Record<string, any> = {
      ENTITY_ID: triggerEvent.entityId,
      ENTITY_TYPE: triggerEvent.entityType,
      TRIGGER_USER: triggerEvent.userId,
      NEW_STAGE: triggerEvent.data.newStageName,
      PREVIOUS_STAGE: triggerEvent.data.previousStageName,
      DEAL_VALUE: triggerEvent.data.newValue,
      TIMESTAMP: triggerEvent.timestamp.toISOString(),
    };
    return tokenMap[token] ?? match;
  });
}
```
